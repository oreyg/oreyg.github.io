38static inline float Distance(const ImVec2& a, const ImVec2& b)
39{
40 return sqrtf((a.x - b.x) * (a.x - b.x) + (a.y - b.y) * (a.y - b.y));
41}
42
43static inline float sign(float v)
44{
45 return (v >= 0.f) ? 1.f : -1.f;
46}

48static ImVec2 GetInputSlotPos(Delegate& delegate, const Node& node, SlotIndex slotIndex, float factor)
49{
50 ImVec2 Size = node.mRect.GetSize() * factor;
51 size_t InputsCount = delegate.GetTemplate(node.mTemplateIndex).mInputCount;
52 return ImVec2(node.mRect.Min.x * factor,
53 node.mRect.Min.y * factor + Size.y * ((float)slotIndex + 1) / ((float)InputsCount + 1) + 8.f);
54}
55
56static ImVec2 GetOutputSlotPos(Delegate& delegate, const Node& node, SlotIndex slotIndex, float factor)
57{
58 ImVec2 Size = node.mRect.GetSize() * factor;
59 size_t OutputsCount = delegate.GetTemplate(node.mTemplateIndex).mOutputCount;
60 return ImVec2(node.mRect.Min.x * factor + Size.x,
61 node.mRect.Min.y * factor + Size.y * ((float)slotIndex + 1) / ((float)OutputsCount + 1) + 8.f);
62}
63
64static ImRect GetNodeRect(const Node& node, float factor)
65{
66 ImVec2 Size = node.mRect.GetSize() * factor;
67 return ImRect(node.mRect.Min * factor, node.mRect.Min * factor + Size);
68}

85static void HandleZoomScroll(ImRect regionRect, ViewState& viewState, const Options& options)
86{
87 ImGuiIO& io = ImGui::GetIO();
88
89 if (regionRect.Contains(io.MousePos))
90 {
91 if (io.MouseWheel < -FLT_EPSILON)
92 {
93 viewState.mFactorTarget *= 1.f - options.mZoomRatio;
94 }
95
96 if (io.MouseWheel > FLT_EPSILON)
97 {
98 viewState.mFactorTarget *= 1.0f + options.mZoomRatio;
99 }
100 }
101
102 ImVec2 mouseWPosPre = (io.MousePos - ImGui::GetCursorScreenPos()) / viewState.mFactor;
103 viewState.mFactorTarget = ImClamp(viewState.mFactorTarget, options.mMinZoom, options.mMaxZoom);
104 viewState.mFactor = ImLerp(viewState.mFactor, viewState.mFactorTarget, options.mZoomLerpFactor);
105 ImVec2 mouseWPosPost = (io.MousePos - ImGui::GetCursorScreenPos()) / viewState.mFactor;
106 if (ImGui::IsMousePosValid())
107 {
108 viewState.mPosition += mouseWPosPost - mouseWPosPre;
109 }
110}

117static void FitNodes(Delegate& delegate, ViewState& viewState, const ImVec2 viewSize, bool selectedNodesOnly)
118{
119 const size_t nodeCount = delegate.GetNodeCount();
120
121 if (!nodeCount)
122 {
123 return;
124 }
125
126 bool validNode = false;
127 ImVec2 min(FLT_MAX, FLT_MAX);
128 ImVec2 max(-FLT_MAX, -FLT_MAX);
129 for (NodeIndex nodeIndex = 0; nodeIndex < nodeCount; nodeIndex++)
130 {
131 const Node& node = delegate.GetNode(nodeIndex);
132
133 if (selectedNodesOnly && !node.mSelected)
134 {
135 continue;
136 }
137
138 min = ImMin(min, node.mRect.Min);
139 min = ImMin(min, node.mRect.Max);
140 max = ImMax(max, node.mRect.Min);
141 max = ImMax(max, node.mRect.Max);
142 validNode = true;
143 }
144
145 if (!validNode)
146 {
147 return;
148 }
149
150 min -= viewSize * 0.05f;
151 max += viewSize * 0.05f;
152 ImVec2 nodesSize = max - min;
153 ImVec2 nodeCenter = (max + min) * 0.5f;
154
155 float ratioY = viewSize.y / nodesSize.y;
156 float ratioX = viewSize.x / nodesSize.x;
157
158 viewState.mFactor = viewState.mFactorTarget = ImMin(ImMin(ratioY, ratioX), 1.f);
159 viewState.mPosition = ImVec2(-nodeCenter.x, -nodeCenter.y) + (viewSize * 0.5f) / viewState.mFactorTarget;
160}

187 {

190 drawList->AddBezierCubic(p1, p1 + ImVec2(50, 0) * factor, p2 + ImVec2(-50, 0) * factor, p2, col, options.mLineThickness * 1.5f * factor);

193 ImVec2 p20 = p2 - ImVec2(20.f * factor, 0.f);

195 ImVec2 dif = p20 - p10;
196 ImVec2 p1a, p1b;
197 if (fabsf(dif.x) > fabsf(dif.y))
198 {
199 p1a = p10 + ImVec2(fabsf(fabsf(dif.x) - fabsf(dif.y)) * 0.5 * sign(dif.x), 0.f);
200 p1b = p1a + ImVec2(fabsf(dif.y) * sign(dif.x) , dif.y);
201 }
202 else

216 // straight lines

220
221 ImVec2 p1a, p1b;
222 const float limitx = 12.f * factor;
223 if (dif.x < limitx)
224 {
225 ImVec2 p10 = p1 + ImVec2(limitx, 0.f);
226 ImVec2 p20 = p2 - ImVec2(limitx, 0.f);
227
228 dif = p20 - p10;
229 p1a = p10 + ImVec2(0.f, dif.y * 0.5f);
230 p1b = p1a + ImVec2(dif.x, 0.f);
231
232 pts = { p1, p10, p1a, p1b, p20, p2 };
233 ptCount = 6;
234 }
235 else
236 {
237 if (fabsf(dif.y) < 1.f)
238 {
239 pts = { p1, (p1 + p2) * 0.5f, p2 };
240 ptCount = 3;
241 }
242 else
243 {
244 if (fabsf(dif.y) < 10.f)
245 {
246 if (fabsf(dif.x) > fabsf(dif.y))
247 {
248 p1a = p1 + ImVec2(fabsf(fabsf(dif.x) - fabsf(dif.y)) * 0.5f * sign(dif.x), 0.f);
249 p1b = p1a + ImVec2(fabsf(dif.y) * sign(dif.x), dif.y);
250 }
251 else
252 {
253 p1a = p1 + ImVec2(0.f, fabsf(fabsf(dif.y) - fabsf(dif.x)) * 0.5f * sign(dif.y));
254 p1b = p1a + ImVec2(dif.x, fabsf(dif.x) * sign(dif.y));
255 }
256 }
257 else
258 {
259 if (fabsf(dif.x) > fabsf(dif.y))
260 {
261 float d = fabsf(dif.y) * sign(dif.x) * 0.5f;
262 p1a = p1 + ImVec2(d, dif.y * 0.5f);
263 p1b = p1a + ImVec2(fabsf(fabsf(dif.x) - fabsf(d) * 2.f) * sign(dif.x), 0.f);
264 }
265 else
266 {
267 float d = fabsf(dif.x) * sign(dif.y) * 0.5f;
268 p1a = p1 + ImVec2(dif.x * 0.5f, d);
269 p1b = p1a + ImVec2(0.f, fabsf(fabsf(dif.y) - fabsf(d) * 2.f) * sign(dif.y));
270 }
271 }
272 pts = { p1, p1a, p1b, p2 };
273 ptCount = 4;
274 }
275 }
276 float highLightFactor = factor * (highlightCons ? 2.0f : 1.f);
277 for (int pass = 0; pass < 2; pass++)
278 {
279 drawList->AddPolyline(pts.data(), ptCount, pass ? col : 0xFF000000, false, (pass ? options.mLineThickness : (options.mLineThickness * 1.5f)) * highLightFactor);
280 }
281 }
282 }

285static void HandleQuadSelection(Delegate& delegate, ImDrawList* drawList, const ImVec2 offset, const float factor, ImRect contentRect, const Options& options)
286{
287 if (!options.mAllowQuadSelection)
288 {
289 return;
290 }
291 ImGuiIO& io = ImGui::GetIO();
292 static ImVec2 quadSelectPos;
293 //auto& nodes = delegate->GetNodes();
294 auto nodeCount = delegate.GetNodeCount();
295
296 if (nodeOperation == NO_QuadSelecting && ImGui::IsWindowFocused())
297 {
298 const ImVec2 bmin = ImMin(quadSelectPos, io.MousePos);
299 const ImVec2 bmax = ImMax(quadSelectPos, io.MousePos);
300 drawList->AddRectFilled(bmin, bmax, options.mQuadSelection, 1.f);
301 drawList->AddRect(bmin, bmax, options.mQuadSelectionBorder, 1.f);
302 if (!io.MouseDown[0])
303 {
304 if (!io.KeyCtrl && !io.KeyShift)
305 {
306 for (size_t nodeIndex = 0; nodeIndex < nodeCount; nodeIndex++)
307 {
308 delegate.SelectNode(nodeIndex, false);
309 }
310 }
311
312 nodeOperation = NO_None;
313 ImRect selectionRect(bmin, bmax);
314 for (int nodeIndex = 0; nodeIndex < nodeCount; nodeIndex++)
315 {
316 const auto node = delegate.GetNode(nodeIndex);
317 ImVec2 nodeRectangleMin = offset + node.mRect.Min * factor;
318 ImVec2 nodeRectangleMax = nodeRectangleMin + node.mRect.GetSize() * factor;
319 if (selectionRect.Overlaps(ImRect(nodeRectangleMin, nodeRectangleMax)))
320 {
321 if (io.KeyCtrl)
322 {
323 delegate.SelectNode(nodeIndex, false);
324 }
325 else
326 {
327 delegate.SelectNode(nodeIndex, true);
328 }
329 }
330 else
331 {
332 if (!io.KeyShift)
333 {
334 delegate.SelectNode(nodeIndex, false);
335 }
336 }
337 }
338 }
339 }
340 else if (nodeOperation == NO_None && io.MouseDown[0] && ImGui::IsWindowFocused() &&
341 contentRect.Contains(io.MousePos))
342 {
343 nodeOperation = NO_QuadSelecting;
344 quadSelectPos = io.MousePos;
345 }
346}

348static bool HandleConnections(ImDrawList* drawList,
349 NodeIndex nodeIndex,
350 const ImVec2 offset,
351 const float factor,
352 Delegate& delegate,
353 const Options& options,
354 bool bDrawOnly,
355 SlotIndex& inputSlotOver,
356 SlotIndex& outputSlotOver,
357 const bool inMinimap)
358{
359 static NodeIndex editingNodeIndex;
360 static SlotIndex editingSlotIndex;
361
362 ImGuiIO& io = ImGui::GetIO();
363 const auto node = delegate.GetNode(nodeIndex);
364 const auto nodeTemplate = delegate.GetTemplate(node.mTemplateIndex);
365 const auto linkCount = delegate.GetLinkCount();
366
367 size_t InputsCount = nodeTemplate.mInputCount;
368 size_t OutputsCount = nodeTemplate.mOutputCount;
369 inputSlotOver = static_cast<GraphEditor::SlotIndex>(-1);
370 outputSlotOver = static_cast<GraphEditor::SlotIndex>(-1);
371
372 // draw/use inputs/outputs
373 bool hoverSlot = false;
374 for (int i = 0; i < 2; i++)
375 {
376 float closestDistance = FLT_MAX;
377 SlotIndex closestConn = static_cast<GraphEditor::SlotIndex>(-1);
378 ImVec2 closestTextPos;
379 ImVec2 closestPos;
380 const size_t slotCount[2] = {InputsCount, OutputsCount};
381
382 for (SlotIndex slotIndex = 0; slotIndex < slotCount[i]; slotIndex++)
383 {
384 const char** con = i ? nodeTemplate.mOutputNames : nodeTemplate.mInputNames;
385 const char* conText = (con && con[slotIndex]) ? con[slotIndex] : "";
386
387 ImVec2 p =
388 offset + (i ? GetOutputSlotPos(delegate, node, slotIndex, factor) : GetInputSlotPos(delegate, node, slotIndex, factor));
389 float distance = Distance(p, io.MousePos);
390 bool overCon = (nodeOperation == NO_None || nodeOperation == NO_EditingLink) &&
391 (distance < options.mNodeSlotRadius * 2.f) && (distance < closestDistance);
392
393
394 ImVec2 textSize;
395 textSize = ImGui::CalcTextSize(conText);
396 ImVec2 textPos =
397 p + ImVec2(-options.mNodeSlotRadius * (i ? -1.f : 1.f) * (overCon ? 3.f : 2.f) - (i ? 0 : textSize.x),
398 -textSize.y / 2);
399
400 ImRect nodeRect = GetNodeRect(node, factor);
401 if (!inMinimap && (overCon || (nodeRect.Contains(io.MousePos - offset) && closestConn == -1 &&
402 (editingInput == (i != 0)) && nodeOperation == NO_EditingLink)))
403 {
404 closestDistance = distance;
405 closestConn = slotIndex;
406 closestTextPos = textPos;
407 closestPos = p;
408
409 if (i)
410 {
411 outputSlotOver = slotIndex;
412 }
413 else
414 {
415 inputSlotOver = slotIndex;
416 }
417 }
418 else
419 {
420 const ImU32* slotColorSource = i ? nodeTemplate.mOutputColors : nodeTemplate.mInputColors;
421 const ImU32 slotColor = slotColorSource ? slotColorSource[slotIndex] : options.mDefaultSlotColor;
422 drawList->AddCircleFilled(p, options.mNodeSlotRadius, IM_COL32(0, 0, 0, 200));
423 drawList->AddCircleFilled(p, options.mNodeSlotRadius * 0.75f, slotColor);
424 if (!options.mDrawIONameOnHover)
425 {
426 drawList->AddText(io.FontDefault, 14, textPos + ImVec2(2, 2), IM_COL32(0, 0, 0, 255), conText);
427 drawList->AddText(io.FontDefault, 14, textPos, IM_COL32(150, 150, 150, 255), conText);
428 }
429 }
430 }
431
432 if (closestConn != -1)
433 {
434 const char** con = i ? nodeTemplate.mOutputNames : nodeTemplate.mInputNames;
435 const char* conText = (con && con[closestConn]) ? con[closestConn] : "";
436 const ImU32* slotColorSource = i ? nodeTemplate.mOutputColors : nodeTemplate.mInputColors;
437 const ImU32 slotColor = slotColorSource ? slotColorSource[closestConn] : options.mDefaultSlotColor;
438 hoverSlot = true;
439 drawList->AddCircleFilled(closestPos, options.mNodeSlotRadius * options.mNodeSlotHoverFactor * 0.75f, IM_COL32(0, 0, 0, 200));
440 drawList->AddCircleFilled(closestPos, options.mNodeSlotRadius * options.mNodeSlotHoverFactor, slotColor);
441 drawList->AddText(io.FontDefault, 16, closestTextPos + ImVec2(1, 1), IM_COL32(0, 0, 0, 255), conText);
442 drawList->AddText(io.FontDefault, 16, closestTextPos, IM_COL32(250, 250, 250, 255), conText);
443 bool inputToOutput = (!editingInput && !i) || (editingInput && i);
444 if (nodeOperation == NO_EditingLink && !io.MouseDown[0] && !bDrawOnly)
445 {
446 if (inputToOutput)
447 {
448 // check loopback
449 Link nl;
450 if (editingInput)
451 nl = Link{nodeIndex, closestConn, editingNodeIndex, editingSlotIndex};
452 else
453 nl = Link{editingNodeIndex, editingSlotIndex, nodeIndex, closestConn};
454
455 if (!delegate.AllowedLink(nl.mOutputNodeIndex, nl.mInputNodeIndex))
456 {
457 break;
458 }
459 bool alreadyExisting = false;
460 for (size_t linkIndex = 0; linkIndex < linkCount; linkIndex++)
461 {
462 const auto link = delegate.GetLink(linkIndex);
463 if (!memcmp(&link, &nl, sizeof(Link)))
464 {
465 alreadyExisting = true;
466 break;
467 }
468 }
469
470 if (!alreadyExisting)
471 {
472 for (int linkIndex = 0; linkIndex < linkCount; linkIndex++)
473 {
474 const auto link = delegate.GetLink(linkIndex);
475 if (link.mOutputNodeIndex == nl.mOutputNodeIndex && link.mOutputSlotIndex == nl.mOutputSlotIndex)
476 {
477 delegate.DelLink(linkIndex);
478
479 break;
480 }
481 }
482
483 delegate.AddLink(nl.mInputNodeIndex, nl.mInputSlotIndex, nl.mOutputNodeIndex, nl.mOutputSlotIndex);
484 }
485 }
486 }
487 // when ImGui::IsWindowHovered() && !ImGui::IsAnyItemActive() is uncommented, one can't click the node
488 // input/output when mouse is over the node itself.
489 if (nodeOperation == NO_None &&
490 /*ImGui::IsWindowHovered() && !ImGui::IsAnyItemActive() &&*/ io.MouseClicked[0] && !bDrawOnly)
491 {
492 nodeOperation = NO_EditingLink;
493 editingInput = i == 0;
494 editingNodeSource = closestPos;
495 editingNodeIndex = nodeIndex;
496 editingSlotIndex = closestConn;
497 if (editingInput)
498 {
499 // remove existing link
500 for (int linkIndex = 0; linkIndex < linkCount; linkIndex++)
501 {
502 const auto link = delegate.GetLink(linkIndex);
503 if (link.mOutputNodeIndex == nodeIndex && link.mOutputSlotIndex == closestConn)
504 {
505 delegate.DelLink(linkIndex);
506 break;
507 }
508 }
509 }
510 }
511 }
512 }
513 return hoverSlot;
514}

516static void DrawGrid(ImDrawList* drawList, ImVec2 windowPos, const ViewState& viewState, const ImVec2 canvasSize, ImU32 gridColor, ImU32 gridColor2, float gridSize)
517{
518 float gridSpace = gridSize * viewState.mFactor;
519 int divx = static_cast<int>(-viewState.mPosition.x / gridSize);
520 int divy = static_cast<int>(-viewState.mPosition.y / gridSize);
521 for (float x = fmodf(viewState.mPosition.x * viewState.mFactor, gridSpace); x < canvasSize.x; x += gridSpace, divx ++)
522 {
523 bool tenth = !(divx % 10);
524 drawList->AddLine(ImVec2(x, 0.0f) + windowPos, ImVec2(x, canvasSize.y) + windowPos, tenth ? gridColor2 : gridColor);
525 }
526 for (float y = fmodf(viewState.mPosition.y * viewState.mFactor, gridSpace); y < canvasSize.y; y += gridSpace, divy ++)
527 {
528 bool tenth = !(divy % 10);
529 drawList->AddLine(ImVec2(0.0f, y) + windowPos, ImVec2(canvasSize.x, y) + windowPos, tenth ? gridColor2 : gridColor);
530 }
531}

718bool DrawMiniMap(ImDrawList* drawList, Delegate& delegate, ViewState& viewState, const Options& options, const ImVec2 windowPos, const ImVec2 canvasSize)
719{
720 if (Distance(options.mMinimap.Min, options.mMinimap.Max) <= FLT_EPSILON)
721 {
722 return false;
723 }
724
725 const size_t nodeCount = delegate.GetNodeCount();
726
727 if (!nodeCount)
728 {
729 return false;
730 }
731
732 ImVec2 min(FLT_MAX, FLT_MAX);
733 ImVec2 max(-FLT_MAX, -FLT_MAX);
734 const ImVec2 margin(50, 50);
735 for (NodeIndex nodeIndex = 0; nodeIndex < nodeCount; nodeIndex++)
736 {
737 const Node& node = delegate.GetNode(nodeIndex);
738 min = ImMin(min, node.mRect.Min - margin);
739 min = ImMin(min, node.mRect.Max + margin);
740 max = ImMax(max, node.mRect.Min - margin);
741 max = ImMax(max, node.mRect.Max + margin);
742 }
743
744 // add view in world space
745 const ImVec2 worldSizeView = canvasSize / viewState.mFactor;
746 const ImVec2 viewMin(-viewState.mPosition.x, -viewState.mPosition.y);
747 const ImVec2 viewMax = viewMin + worldSizeView;
748 min = ImMin(min, viewMin);
749 max = ImMax(max, viewMax);
750 const ImVec2 nodesSize = max - min;
751 const ImVec2 middleWorld = (min + max) * 0.5f;
752 const ImVec2 minScreen = windowPos + options.mMinimap.Min * canvasSize;
753 const ImVec2 maxScreen = windowPos + options.mMinimap.Max * canvasSize;
754 const ImVec2 viewSize = maxScreen - minScreen;
755 const ImVec2 middleScreen = (minScreen + maxScreen) * 0.5f;
756 const float ratioY = viewSize.y / nodesSize.y;
757 const float ratioX = viewSize.x / nodesSize.x;
758 const float factor = ImMin(ImMin(ratioY, ratioX), 1.f);
759
760 drawList->AddRectFilled(minScreen, maxScreen, IM_COL32(30, 30, 30, 200), 3, ImDrawFlags_RoundCornersAll);
761
762 for (NodeIndex nodeIndex = 0; nodeIndex < nodeCount; nodeIndex++)
763 {
764 const Node& node = delegate.GetNode(nodeIndex);
765 const auto nodeTemplate = delegate.GetTemplate(node.mTemplateIndex);
766
767 ImRect rect = node.mRect;
768 rect.Min -= middleWorld;
769 rect.Min *= factor;
770 rect.Min += middleScreen;
771
772 rect.Max -= middleWorld;
773 rect.Max *= factor;
774 rect.Max += middleScreen;
775
776 drawList->AddRectFilled(rect.Min, rect.Max, nodeTemplate.mBackgroundColor, 1, ImDrawFlags_RoundCornersAll);
777 if (node.mSelected)
778 {
779 drawList->AddRect(rect.Min, rect.Max, options.mSelectedNodeBorderColor, 1, ImDrawFlags_RoundCornersAll);
780 }
781 }
782
783 // add view
784 ImVec2 viewMinScreen = (viewMin - middleWorld) * factor + middleScreen;
785 ImVec2 viewMaxScreen = (viewMax - middleWorld) * factor + middleScreen;
786 drawList->AddRectFilled(viewMinScreen, viewMaxScreen, IM_COL32(255, 255, 255, 32), 1, ImDrawFlags_RoundCornersAll);
787 drawList->AddRect(viewMinScreen, viewMaxScreen, IM_COL32(255, 255, 255, 128), 1, ImDrawFlags_RoundCornersAll);
788
789 ImGuiIO& io = ImGui::GetIO();
790 const bool mouseInMinimap = ImRect(minScreen, maxScreen).Contains(io.MousePos);
791 if (mouseInMinimap && io.MouseClicked[0])
792 {
793 const ImVec2 clickedRatio = (io.MousePos - minScreen) / viewSize;
794 const ImVec2 worldPosCenter = ImVec2(ImLerp(min.x, max.x, clickedRatio.x), ImLerp(min.y, max.y, clickedRatio.y));
795
796 ImVec2 worldPosViewMin = worldPosCenter - worldSizeView * 0.5;
797 ImVec2 worldPosViewMax = worldPosCenter + worldSizeView * 0.5;
798 if (worldPosViewMin.x < min.x)
799 {
800 worldPosViewMin.x = min.x;
801 worldPosViewMax.x = worldPosViewMin.x + worldSizeView.x;
802 }
803 if (worldPosViewMin.y < min.y)
804 {
805 worldPosViewMin.y = min.y;
806 worldPosViewMax.y = worldPosViewMin.y + worldSizeView.y;
807 }
808 if (worldPosViewMax.x > max.x)
809 {
810 worldPosViewMax.x = max.x;
811 worldPosViewMin.x = worldPosViewMax.x - worldSizeView.x;
812 }
813 if (worldPosViewMax.y > max.y)
814 {
815 worldPosViewMax.y = max.y;
816 worldPosViewMin.y = worldPosViewMax.y - worldSizeView.y;
817 }
818 viewState.mPosition = ImVec2(-worldPosViewMin.x, -worldPosViewMin.y);
819 }
820 return mouseInMinimap;
821}
