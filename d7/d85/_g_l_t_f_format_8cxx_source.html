43 static int hq_gltf_size(std::string_view string, int type)
44 {
45 using namespace ComponentType;
46 int c = 1;
47 switch(type) {
48 case SHORT:
49 case UNSIGNED_SHORT:
50 c = 2;
51 break;
52 case UNSIGNED_INT:
53 case FLOAT:
54 c = 4;
55 break;
56 }
57
58 if (string == "SCALAR") {
59 return 1 * c;
60 }
61 else if (string == "VEC2") {
62 return 2 * c;
63 }
64 else if (string == "VEC3") {
65 return 3 * c;
66 }
67 else if (string == "VEC4") {
68 return 4 * c;
69 }
70 return 0;
71 }

101 GLTFExtracted GLTFFormat::Extract(int index) const
102 {
103 GLTFExtracted result{};
104 if (index == -1)
105 {
106 return result;
107 }
108
109 const auto& accessor = m_root.accessors[index];
110 const auto& bufferView = m_root.bufferViews[accessor.bufferView];
111 const auto* bufferData = m_buffers[bufferView.buffer].data();
112
113 result.count = accessor.count;
114 result.type = accessor.componentType;
115 result.data = bufferData + bufferView.byteOffset + accessor.byteOffset;
116 result.stride = bufferView.byteStride > 0 ? bufferView.byteStride : hq_gltf_size(accessor.type, accessor.componentType);
117 return result;
118 }

183 void GLTFFormat::LoadFromFile(std::string_view path)
184 {
185 ZoneScoped;
186
187 Path p{ path };
188
189 hqmeta::InputJSONArchive archive{ hqfs::ReadTextFile(p.data()) };
190 hqmeta::Serialize(archive, m_root);
191
192 for (const auto& buffer : m_root.buffers)
193 {
194 std::string subpath{ hqpath::filepath(path) };
195 subpath += buffer.uri;
196 m_buffers.emplace_back( hqfs::ReadBulkFile(subpath.c_str()) );
197 }
198
199 for (const auto& mesh : m_root.meshes)
200 {
201 for (const auto& primitive : mesh.primitives)
202 {
203 if (m_primitives.has(primitive.indices))
204 {
205 return;
206 }
207
208 hq::FVertexArray vertexArray;
209 hq::FVIndexArray vindexArray;
210
211 auto position = Extract(primitive.attributes.POSITION);
212 auto normal = Extract(primitive.attributes.NORMAL);
213 auto color0 = Extract(primitive.attributes.COLOR_0);
214 auto texcoord0 = Extract(primitive.attributes.TEXCOORD_0);
215 auto tangent = Extract(primitive.attributes.TANGENT);
216
217 if (!IsVecX(primitive.attributes.POSITION)) position = {};
218 if (!IsVecX(primitive.attributes.NORMAL)) normal = {};
219 if (!IsVecX(primitive.attributes.COLOR_0)) color0 = {};
220 if (!IsVec2(primitive.attributes.TEXCOORD_0)) texcoord0 = {};
221 if (!IsVecX(primitive.attributes.TANGENT)) tangent = {};
222
223 bool validIndices = IsU16(primitive.indices) || IsU32(primitive.indices);
224 const size_t vertSize = position.count;
225 if (position.count != normal.count ||
226 position.count != texcoord0.count ||
227 !validIndices)
228 {
229 HLOG_ERROR(GLTF, "Failed to construct a primitive: size of attributes does not match.");
230 continue;
231 }
232
233 vertexArray.reserve(vertSize);
234 for (u32 i = 0; i < position.count; ++i)
235 {
236 hq::FVertexLayout layout;
237 memset(&layout, 0, sizeof(hq::FVertexLayout));
238 if (position.data)
239 {
240 memcpy(layout.position, position.data, sizeof(float3));
241 position.data += position.stride;
242 }
243 if (normal.data)
244 {
245 memcpy(layout.normal, normal.data, sizeof(float3));
246 normal.data += normal.stride;
247 }
248 if (color0.data)
249 {
250 memcpy(layout.color, color0.data, sizeof(float3));
251 color0.data += color0.stride;
252 }
253 if (texcoord0.data)
254 {
255 memcpy(layout.texcord, texcoord0.data, sizeof(float2));
256 texcoord0.data += texcoord0.stride;
257 }
258 if (tangent.data)
259 {
260 memcpy(layout.tangent, tangent.data, sizeof(float3));
261 tangent.data += tangent.stride;
262 }
263 vertexArray.emplace_back(layout);
264 }
265
266 auto indices = Extract(primitive.indices);
267 vindexArray.resize(indices.count);
268 if (indices.stride == sizeof(u16))
269 {
270 for (uint32_t i = 0; i < indices.count; ++i)
271 {
272 vindexArray[i] = *reinterpret_cast<const u16*>(indices.data);
273 indices.data += indices.stride;
274 }
275 }
276 else
277 {
278 for (uint32_t i = 0; i < indices.count; ++i)
279 {
280 vindexArray[i] = *reinterpret_cast<const u32*>(indices.data);
281 indices.data += indices.stride;
282 }
283 }
284
285 m_primitives.insert( primitive.indices, std::move(vertexArray), std::move(vindexArray), mesh.name);
286 }
287 }
288
289 m_materials.resize(0);
290 m_materials.reserve(m_root.materials.size());
291 for (const auto& material : m_root.materials)
292 {
293 hq::FRenderMaterialParams params;
294
295 params.m_name = material.name;
296 params.normal.name = GetImagePath(path.data(), material.normalTexture.index);
297 params.baseColor.name = GetImagePath(path.data(), material.pbrMetallicRoughness.baseColorTexture.index);
298 params.metallicRoughness.name = GetImagePath(path.data(), material.pbrMetallicRoughness.metallicRoughnessTexture.index);
299 params.emissive.name = GetImagePath(path.data(), material.emissiveTexture.index);
300 params.renderType = 3;
301 params.normal.isGenMips = true;
302 params.baseColor.isGenMips = true;
303 params.metallicRoughness.isGenMips = true;
304
305 m_materials.emplace_back(std::move(params));
306 }
