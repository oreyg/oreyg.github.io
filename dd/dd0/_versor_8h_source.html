64 HQINLINE FVersor(FEuler r) noexcept
65 {
66 const FEuler hr = r * 0.5f;
67 const float sinx = Sin(hr.x);
68 const float cosx = Cos(hr.x);
69 const float siny = Sin(hr.y);
70 const float cosy = Cos(hr.y);
71 const float sinz = Sin(hr.z);
72 const float cosz = Cos(hr.z);
73 x = sinx * cosy * cosz - cosx * siny * sinz;
74 y = cosx * siny * cosz + sinx * cosy * sinz;
75 z = cosx * cosy * sinz - sinx * siny * cosz;
76 w = cosx * cosy * cosz + sinx * siny * sinz;
77 }

98 HQINLINE FEuler ToEuler() const noexcept
99 {
100 float t0 = 2.0f * (w * x + y * z);
101 float t1 = 1.0f - 2.0f * (x * x + y * y);
102 float roll = ATan2(t0, t1);
103 float t2 = Clamp(2.0f * (w * y - z * x), -1.0f, 1.0f);
104 float pitch = ASin(t2);
105 float t3 = 2.0f * (w * z + x * y);
106 float t4 = 1.0f - 2.0f * (y * y + z * z);
107 float yaw = ATan2(t3, t4);
108 return { roll, pitch, yaw };
109 }

266 {

268 constexpr float re = 1.0f - e;

279 inline FVersor MakeRotator(FVector3 from, FVector3 to) noexcept
280 {
281 FVector3 forward = math::Normalize(from);
282 FVector3 direction = math::Normalize(to);
283
284 float cosTheta = math::Dot(forward, direction);
285 FVector3 axis;
286
287 constexpr float e = 1.e-6f;
288 if (cosTheta < e - 1.f) {
289 // special case when vectors in opposite directions:
290 // there is no "ideal" rotation axis
291 // So guess one; any will do as long as it's perpendicular to start
292 axis = math::Cross(FVector::Z_AXIS(), forward);
293 if (math::LengthSqr(axis) < 0.01f)
294 {
295 axis = math::Cross( FVector::Y_AXIS(), forward);
296 }
297
298 axis = math::Normalize(axis);
299 return FVersor(axis.x, axis.y, axis.z, 0);
300 }
301
302 axis = math::Cross(forward, direction);
303 float s = sqrtf((1.f + cosTheta) * 2.f);
304 float rcp = 1.f / s;
305
306 FVersor result(
307 axis.x * rcp,
308 axis.y * rcp,
309 axis.z * rcp,
310 s * 0.5f
311 );
312
313 return Normalize(result);
314 }
