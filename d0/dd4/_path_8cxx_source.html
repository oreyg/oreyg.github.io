7void Path::normalize()
8{
9 if (m_path.empty())
10 {
11 return;
12 }
13
14 // Convert all backward slashes into forward slashes
15 hqstring::replace(m_path, '\\', '/');
16
17 size_t s0 = 0;
18 size_t s1 = 0;
19 size_t s2 = 0;
20
21 // Skip first token if path is absolute
22 if (!is_relative())
23 {
24 s2 = hqstring::ffind(m_path, '/');
25 s1 = s2 + 1;
26 }
27 else
28 {
29 // Make sure that our path starts with /
30 if (!hqstring::starts_with(m_path, '/'))
31 {
32 m_path.insert(0, 1, '/');
33 }
34 }
35
36 bool endsWithSlash = hqstring::ends_with(m_path, '/');
37
38 // Apply all ..
39 while (hqstring::found(m_path, s2))
40 {
41 s0 = s1;
42 s1 = s2;
43 // Skip all continuous slashes
44 while (utf8::is_first_cu(m_path[s1]) && m_path[s1] == '/')
45 {
46 ++s1;
47 }
48
49 // Find next packs of slashes
50 s2 = hqstring::ffind(m_path, '/', s1 + 1);
51 if (s0 == s2)
52 {
53 return;
54 }
55
56 // If we have .. then remove it and previous token
57 std::string_view view{ &m_path[s1], s2 - s1 };
58 if (view == "..")
59 {
60 m_path.erase(s0, s2 - s0);
61 s1 = s0;
62 s2 = s1;
63 }
64 }
65
66 // Remove all ../ from the beginning
67 if (hqstring::starts_with(m_path, "../"))
68 {
69 constexpr size_t toRemove = sizeof("../") - 1;
70 m_path.erase(0, toRemove);
71 }
72
73 // Remove all ./ from the beginning
74 if (!hqstring::ends_with(m_path, '/') && endsWithSlash)
75 {
76 m_path.append('/', 1);
77 }
78
79 // Replace all // with /
80 hqstring::replace(m_path, "//", '/');
81
82 // Transform to lower
83 hqstring::to_lower(m_path);
84}
