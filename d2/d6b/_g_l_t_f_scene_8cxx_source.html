41 HGLScene GLTFSceneSystem::LoadScene(const Path& path)
42 {
43 size_t* foundIndex = m_mapping.get(path);
44 if (foundIndex)
45 {
46 return CInvalidGLScene;
47 }
48
49 auto newAsset = m_assets.emit();
50 size_t resultingIndex = std::distance(m_assets.data(), newAsset.get());
51
52 auto& gltf = newAsset->m_gltf;
53 gltf.LoadFromFile(path);
54
55 newAsset->m_rendMat.resize(gltf.GetMaterials().size());
56 for (uint32_t i = 0; i < newAsset->m_rendMat.size(); ++i)
57 {
58 newAsset->m_rendMat[i] = hq::GFX_CreateMaterial(gltf.GetMaterials()[i]);
59 }
60
61 for (const auto& primitive : gltf.GetPrimitives())
62 {
63 newAsset->m_rendMsh.set(primitive.first, hq::GFX_CreateMesh(primitive.second));
64 }
65
66 auto& root = gltf.GetRoot();
67 auto sceneTemplate = m_scenes.emit();
68 for (const auto& node : root.nodes)
69 {
70 for (auto& child : node.children)
71 {
72 if (child == -1)
73 {
74 HLOG_ERROR(GLTFScene, "Missing child %s", path.c_str());
75 continue;
76 }
77 FTransform parentTransform;
78 parentTransform.position = node.translation;
79 parentTransform.rotation = node.rotation;
80 FTransform childTransform;
81 childTransform.position = root.nodes[child].translation;
82 childTransform.rotation = root.nodes[child].rotation;
83 parentTransform.Apply(childTransform);
84 memcpy(const_cast<float*>(root.nodes[child].translation), &parentTransform.position, sizeof(float[3]));
85 memcpy(const_cast<float*>(root.nodes[child].rotation), &parentTransform.rotation, sizeof(float[4]));
86 }
87 if (node.mesh == -1)
88 {
89 continue;
90 }
91
92 sceneTemplate->m_nodes.emplace_back();
93 auto& newNode = sceneTemplate->m_nodes.back();
94
95 FTransform transform;
96 transform.position = node.translation;
97 transform.rotation = node.rotation;
98 newNode.transform = transform;
99
100 auto& mesh = root.meshes[node.mesh];
101 for (uint32_t primIndex = 0; primIndex < mesh.primitives.size(); ++primIndex)
102 {
103 std::string primName{ std::to_string(primIndex).c_str() };
104 primName.append(node.name);
105
106 auto& primitive = mesh.primitives[primIndex];
107 hq::HResHandle* msh = newAsset->m_rendMsh.get(primitive.indices);
108 hq::HResHandle mat = newAsset->m_rendMat[primitive.material];
109 if (msh != nullptr)
110 {
111 newNode.mesh.primitives.emplace_back(*msh, mat, primName);
112 }
113 }
114 }
115
116 m_mapping.set(path, resultingIndex);
117 return HGLScene{ resultingIndex };
118 }

132 HGLInst GLTFSceneSystem::CreateInstance(HGLScene scene, const math::FTransform& transform, hq::RenderPassMask passMask)
133 {
134 size_t sceneIndex = static_cast<size_t>(scene);
135 if (!m_assets.alive(sceneIndex))
136 {
137 return CInvalidGLInst;
138 }
139
140 auto newInstance = m_instances.emit();
141 GLTFScene* gltfScene = &m_scenes[sceneIndex];
142 for (const auto& node : gltfScene->m_nodes)
143 {
144 for (const auto& prim : node.mesh.primitives)
145 {
146 auto meshHandle = prim.meshHandle;
147 auto materialHandle = prim.materialHandle;
148 entt::entity entity = (*m_scene)->create();
149
150 FTransform t = node.transform;
151 t.Apply(transform);
152 (*m_scene)->emplace<world::TransformComponent>(entity, t);
153 (*m_scene)->emplace<world::NamebleComponent>(entity, prim.name);
154 (*m_scene)->emplace<world::SelectableComponent>(entity, false);
155 m_renderSystem->AttachMeshComponent(entity, meshHandle, materialHandle, passMask);
156 newInstance->m_entities.push_back(entity);
157 }
158 }
159
160 size_t instanceIndex = std::distance(m_instances.data(), newInstance.get());
161 return HGLInst{ instanceIndex };
162 }
