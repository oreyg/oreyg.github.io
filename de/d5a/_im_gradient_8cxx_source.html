32#ifndef IMGUI_DEFINE_MATH_OPERATORS
33 static inline ImVec2 operator*(const ImVec2& lhs, const float rhs) { return ImVec2(lhs.x * rhs, lhs.y * rhs); }
34 static inline ImVec2 operator/(const ImVec2& lhs, const float rhs) { return ImVec2(lhs.x / rhs, lhs.y / rhs); }
35 static inline ImVec2 operator+(const ImVec2& lhs, const ImVec2& rhs) { return ImVec2(lhs.x + rhs.x, lhs.y + rhs.y); }
36 static inline ImVec2 operator-(const ImVec2& lhs, const ImVec2& rhs) { return ImVec2(lhs.x - rhs.x, lhs.y - rhs.y); }
37 static inline ImVec2 operator*(const ImVec2& lhs, const ImVec2& rhs) { return ImVec2(lhs.x * rhs.x, lhs.y * rhs.y); }
38 static inline ImVec2 operator/(const ImVec2& lhs, const ImVec2& rhs) { return ImVec2(lhs.x / rhs.x, lhs.y / rhs.y); }

41 static int DrawPoint(ImDrawList* draw_list, ImVec4 color, const ImVec2 size, bool editing, ImVec2 pos)
42 {
43 ImGuiIO& io = ImGui::GetIO();
44
45 ImVec2 p1 = ImLerp(pos, ImVec2(pos + ImVec2(size.x - size.y, 0.f)), color.w) + ImVec2(3, 3);
46 ImVec2 p2 = ImLerp(pos + ImVec2(size.y, size.y), ImVec2(pos + size), color.w) - ImVec2(3, 3);
47 ImRect rc(p1, p2);
48
49 color.w = 1.f;
50 draw_list->AddRectFilled(p1, p2, ImColor(color));
51 if (editing)
52 draw_list->AddRect(p1, p2, 0xFFFFFFFF, 2.f, 15, 2.5f);
53 else
54 draw_list->AddRect(p1, p2, 0x80FFFFFF, 2.f, 15, 1.25f);
55
56 if (rc.Contains(io.MousePos))
57 {
58 if (io.MouseClicked[0])
59 return 2;
60 return 1;
61 }
62 return 0;
63 }

65 bool Edit(Delegate& delegate, const ImVec2& size, int& selection)
66 {
67 bool ret = false;
68 ImGuiIO& io = ImGui::GetIO();
69 ImGui::PushStyleVar(ImGuiStyleVar_FramePadding, ImVec2(0, 0));
70 ImGui::BeginChildFrame(137, size);
71
72 ImDrawList* draw_list = ImGui::GetWindowDrawList();
73 const ImVec2 offset = ImGui::GetCursorScreenPos();
74
75 const ImVec4* pts = delegate.GetPoints();
76 static int currentSelection = -1;
77 static int movingPt = -1;
78 if (currentSelection >= int(delegate.GetPointCount()))
79 currentSelection = -1;
80 if (movingPt != -1)
81 {
82 ImVec4 current = pts[movingPt];
83 current.w += io.MouseDelta.x / size.x;
84 current.w = ImClamp(current.w, 0.f, 1.f);
85 delegate.EditPoint(movingPt, current);
86 ret = true;
87 if (!io.MouseDown[0])
88 movingPt = -1;
89 }
90 for (size_t i = 0; i < delegate.GetPointCount(); i++)
91 {
92 int ptSel = DrawPoint(draw_list, pts[i], size, i == currentSelection, offset);
93 if (ptSel == 2)
94 {
95 currentSelection = int(i);
96 ret = true;
97 }
98 if (ptSel == 1 && io.MouseDown[0] && movingPt == -1)
99 {
100 movingPt = int(i);
101 }
102 }
103 ImRect rc(offset, offset + size);
104 if (rc.Contains(io.MousePos) && io.MouseDoubleClicked[0])
105 {
106 float t = (io.MousePos.x - offset.x) / size.x;
107 delegate.AddPoint(delegate.GetPoint(t));
108 ret = true;
109 }
110 ImGui::EndChildFrame();
111 ImGui::PopStyleVar();
112
113 selection = currentSelection;
114 return ret;
115 }
