435template <typename T, typename Tag>
436struct SafeType
437{
438 SafeType(T t)
439 : m_Value(std::move(t))
440 {
441 }
442
443 SafeType(const SafeType&) = default;
444
445 template <typename T2, typename Tag2>
446 SafeType(
447 const SafeType
448 <
449 typename std::enable_if<!std::is_same<T, T2>::value, T2>::type,
450 typename std::enable_if<!std::is_same<Tag, Tag2>::value, Tag2>::type
451 >&) = delete;
452
453 SafeType& operator=(const SafeType&) = default;
454
455 explicit operator T() const { return Get(); }
456
457 T Get() const { return m_Value; }
458
459private:
460 T m_Value;
461};

464template <typename Tag>
465struct SafePointerType
466 : SafeType<uintptr_t, Tag>
467{
468 static const Tag Invalid;
469
470 using SafeType<uintptr_t, Tag>::SafeType;
471
472 SafePointerType()
473 : SafePointerType(Invalid)
474 {
475 }
476
477 template <typename T = void> explicit SafePointerType(T* ptr): SafePointerType(reinterpret_cast<uintptr_t>(ptr)) {}
478 template <typename T = void> T* AsPointer() const { return reinterpret_cast<T*>(this->Get()); }
479
480 explicit operator bool() const { return *this != Invalid; }
481};
482
483template <typename Tag>
484const Tag SafePointerType<Tag>::Invalid = { 0 };
485
486template <typename Tag>
487inline bool operator==(const SafePointerType<Tag>& lhs, const SafePointerType<Tag>& rhs)
488{
489 return lhs.Get() == rhs.Get();
490}
491
492template <typename Tag>
493inline bool operator!=(const SafePointerType<Tag>& lhs, const SafePointerType<Tag>& rhs)
494{
495 return lhs.Get() != rhs.Get();
496}
