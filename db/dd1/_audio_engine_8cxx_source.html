82 void LoadSound(const Path& soundPath, bool is3d, bool isLooping, bool stream)
83 {
84 FMOD::Sound** foundSound = m_sounds.get(soundPath);
85 if (foundSound)
86 {
87 return;
88 }
89
90 // TEMPSHIT
91 Path resolvedPath{ hqfs::GetBaseDir() };
92 resolvedPath.append("/");
93 resolvedPath.append(soundPath);
94 resolvedPath.normalize();
95 const char* resolvedCStr = resolvedPath.data();
96
97 FMOD_MODE mode = FMOD_DEFAULT;
98 mode |= is3d ? FMOD_3D : FMOD_2D;
99 mode |= isLooping ? FMOD_LOOP_NORMAL : FMOD_LOOP_OFF;
100 mode |= stream ? FMOD_CREATESTREAM : FMOD_CREATECOMPRESSEDSAMPLE;
101 FMOD::Sound* sound = nullptr;
102 FMOD_RESULT result = m_system->createSound(resolvedCStr, mode, nullptr, &sound);
103 if (result == FMOD_OK && sound)
104 {
105 m_sounds.insert(soundPath, sound);
106 }
107 }

131 int PlaySound(const Path& soundPath, const math::FVector3& position, float volume, bool is3d, bool isLooping)
132 {
133 UNREFERENCED(position);
134 UNREFERENCED(volume);
135
136 FMOD::Sound** foundSound = m_sounds.get(soundPath);
137 if (!foundSound)
138 {
139 LoadSound(soundPath, is3d, isLooping, false);
140 foundSound = m_sounds.get(soundPath);
141 if (!foundSound)
142 {
143 return 0;
144 }
145 }
146
147 FMOD::Channel* pChannel = nullptr;
148 FMOD_RESULT result = m_system->playSound(*foundSound, nullptr, true, &pChannel);
149 if (result == FMOD_OK && pChannel)
150 {
151 FMOD_VECTOR pos = VectorToFMOD(position);
152 FMOD_VECTOR vel{};
153 pChannel->setVolume(volume);
154 pChannel->set3DAttributes(&pos, &vel);
155 pChannel->setLoopCount(isLooping ? -1 : 0);
156
157 FMOD_MODE mode{};
158 pChannel->getMode(&mode);
159
160 if (is3d)
161 {
162 mode &= ~FMOD_2D;
163 mode |= FMOD_3D;
164 }
165 else
166 {
167 mode &= ~FMOD_3D;
168 mode |= FMOD_2D;
169 }
170
171 if (isLooping)
172 {
173 mode &= ~FMOD_LOOP_OFF;
174 mode |= FMOD_LOOP_NORMAL;
175 }
176 else
177 {
178 mode &= ~FMOD_LOOP_NORMAL;
179 mode |= FMOD_LOOP_OFF;
180 }
181
182 pChannel->setMode(mode);
183 pChannel->setPaused(false);
184 auto emitChannel = m_channels.emit(pChannel);
185 return (int) std::distance(m_channels.data(), emitChannel.get());
186 }
187 return 0;
188 }
