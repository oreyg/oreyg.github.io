80 void AudioSystem::UpdateAudioComponents(float deltaTime)
81 {
82 ZoneScoped;
83 UNREFERENCED(deltaTime);
84
85 auto view = (*m_scene)->view<world::TransformComponent, AudioComponent>();
86 for (auto [entity, t, audio] : view.each())
87 {
88 //std::unique_lock lock{ audio.mtx };
89
90 // Check that we have a pending sound that is still not loaded
91 // If that is the case, skip this entity
92 if (!audio.rsoundLoaded && audio.oldSound.IsValid())
93 {
94 if (m_resourceManager->IsLoaded(audio.oldSound))
95 {
96 audio.rsoundLoaded = true;
97 }
98 else
99 {
100 continue;
101 }
102 }
103
104 // Check that we have a sound to load
105 if (audio.rsoundChanged)
106 {
107 if (audio.oldSound.IsValid())
108 {
109 m_resourceManager->RequestUnload(audio.oldSound);
110 }
111
112 audio.oldSound = audio.newSound;
113 m_resourceManager->RequestLoad(audio.oldSound);
114 audio.rsoundChanged = false;
115 audio.rsoundLoaded = false;
116 continue;
117 }
118
119 // This entity never requested a sound
120 if (!audio.oldSound.IsValid())
121 {
122 continue;
123 }
124
125 // Update position
126 math::FVector3 position = t.transform.position;
127 if (audio.channel != -1)
128 {
129 m_audioEngine->SetChannelPosition(audio.channel, position);
130 audio.isPlaying = m_audioEngine->IsPlaying(audio.channel);
131
132 if (audio.requestedStop)
133 {
134 m_audioEngine->StopChannel(audio.channel);
135 audio.requestedStop = false;
136 audio.isPlaying = false;
137 }
138 else if (audio.volumeChanged)
139 {
140 m_audioEngine->SetChannelVolume(audio.channel, audio.volume);
141 audio.volumeChanged = false;
142 }
143 else if (audio.modeChanged)
144 {
145 bool new3D = audio.is3D;
146 bool newLp = audio.isLooping;
147 m_audioEngine->ChannelMode(audio.channel, &new3D, &newLp);
148 audio.modeChanged = false;
149 }
150 }
151
152 if (audio.requestedPlay)
153 {
154 if (audio.channel != -1)
155 {
156 m_audioEngine->StopChannel(audio.channel);
157 }
158
159 // Get loaded sound
160 auto* resourceData = m_resourceManager->GetData(audio.oldSound);
161 const Path& path = resourceData->GetPath();
162
163 audio.channel = m_audioEngine->PlaySound(path, position, audio.volume, audio.is3D, audio.isLooping);
164 audio.requestedPlay = false;
165
166 audio.is3D = audio.is3D;
167 audio.isLooping = audio.isLooping;
168 audio.modeChanged = false;
169 }
170 }
171
